## array2Tree

```js
function array2tree({ key, pKey, data }) {
  let obj = {};
  let result = [];
  // 生成map
  obj = data.reduce((acc, current) => {
    acc[current[key]] = current;
    return acc;
  }, {});
  data.forEach((item) => {
    // 获取父节点
    let pNode = obj[item[pKey]];
    if (pNode) {
      if (pNode.children) {
        pNode.children.push(item);
      } else {
        pNode.children = [item];
      }
    } else {
      result.push(item);
    }
  });
  return result;
}
//
let arr = [{ id: 1 },{ id: 2, pId: 1 },{ id: 3, pId: 2 },{ id: 4 },{ id: 5, pId: 2 },{ id: 6, pId: 4 }];
let rs = array2tree({ key: "id", pKey: "pId", data: arr });
console.log(rs);
```

## tree2Array

```js
function tree2array(arr) {
  let result = [];
  let next = function (list) {
    list.forEach((item) => {
      if (item.children) {
        next(item.children);
      }
      delete item.children;
      result.push(item);
    });
  };
  next(arr);
  return result;
}
//
let rs2 = tree2array(rs);
console.log(rs2);
```
