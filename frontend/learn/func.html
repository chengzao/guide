    # 函数自调用

    -函数可以调用自身,这就是递归(recursion)

    function fib(num) {
  if (num === 0) return 0;
  if (num === 1) return 1;
  return fib(num - 2) + fib(num - 1);
}
fib(6); // 8

    # 闭包

    • 闭包的最大用处有两个,
    • 一个是可以读取函数内部的变量
    • 另一个就是让这些变量始终保持在内存中,即闭包可以使得它诞生环境一直存在
    function createIncrementor(start) {
  return function() {
    return start++;
  };
}
var inc = createIncrementor(5);
inc(); // 5
inc(); // 6
inc(); // 7

    • 闭包的另一个用处,是封装对象的私有属性和私有方法
    function Person(name) {
  var _age;
  function setAge(n) {
    _age = n;
  }
  function getAge() {
    return _age;
  }
  return {
    name: name,
    getAge: getAge,
    setAge: setAge
  };
}
var p1 = Person("张三");
p1.setAge(25);
p1.getAge(); // 25
